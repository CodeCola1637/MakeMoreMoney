```python
# strategy/data_normalizer.py
import numpy as np
import pickle
import os

class DataNormalizer:
    """统一的数据归一化器"""
    
    def __init__(self, scaler_path="./models/scaler.pkl"):
        self.scaler_path = scaler_path
        self.feature_ranges = {}  # {feature_idx: (min, max)}
        
    def fit(self, data: np.ndarray, feature_names: list):
        """训练时拟合归一化参数"""
        for i, name in enumerate(feature_names):
            self.feature_ranges[i] = {
                'min': float(data[:, i].min()),
                'max': float(data[:, i].max()),
                'name': name
            }
        self.save()
        
    def transform(self, data: np.ndarray) -> np.ndarray:
        """使用保存的参数进行归一化"""
        normalized = np.zeros_like(data, dtype=np.float32)
        
        for i in range(data.shape[1]):
            if i in self.feature_ranges:
                min_val = self.feature_ranges[i]['min']
                max_val = self.feature_ranges[i]['max']
                if max_val > min_val:
                    normalized[:, i] = (data[:, i] - min_val) / (max_val - min_val)
                else:
                    normalized[:, i] = 0.5
            else:
                # 未知特征使用局部归一化
                min_val = data[:, i].min()
                max_val = data[:, i].max()
                if max_val > min_val:
                    normalized[:, i] = (data[:, i] - min_val) / (max_val - min_val)
                else:
                    normalized[:, i] = 0.5
                    
        return normalized
    
    def save(self):
        """保存归一化参数"""
        with open(self.scaler_path, 'wb') as f:
            pickle.dump(self.feature_ranges, f)
            
    def load(self):
        """加载归一化参数"""
        if os.path.exists(self.scaler_path):
            with open(self.scaler_path, 'rb') as f:
                self.feature_ranges = pickle.load(f)
            return True
        return False
```

```python
# strategy/feature_engineer.py
import pandas as pd
import numpy as np

class FeatureEngineer:
    """特征工程器"""
    
    def __init__(self, config):
        self.config = config
        self.features = config.get("strategy.training.features", 
                                   ["close", "volume", "high", "low"])
    
    def add_technical_indicators(self, df: pd.DataFrame) -> pd.DataFrame:
        """添加技术指标特征"""
        df = df.copy()
        
        # RSI
        df['rsi'] = self._calculate_rsi(df['close'], period=14)
        
        # MACD
        df['macd'], df['macd_signal'], df['macd_hist'] = self._calculate_macd(df['close'])
        
        # 布林带
        df['bb_upper'], df['bb_middle'], df['bb_lower'] = self._calculate_bollinger(df['close'])
        df['bb_position'] = (df['close'] - df['bb_lower']) / (df['bb_upper'] - df['bb_lower'])
        
        # 移动平均
        df['sma_5'] = df['close'].rolling(5).mean()
        df['sma_20'] = df['close'].rolling(20).mean()
        df['ema_12'] = df['close'].ewm(span=12).mean()
        
        # 价格变化率
        df['price_change_1d'] = df['close'].pct_change(1)
        df['price_change_5d'] = df['close'].pct_change(5)
        
        # 成交量特征
        df['volume_ma'] = df['volume'].rolling(20).mean()
        df['volume_ratio'] = df['volume'] / df['volume_ma']
        
        # 波动率
        df['volatility'] = df['close'].rolling(20).std() / df['close'].rolling(20).mean()
        
        # 填充NaN
        df = df.fillna(method='bfill').fillna(method='ffill')
        
        return df
    
    def _calculate_rsi(self, prices, period=14):
        delta = prices.diff()
        gain = (delta.where(delta > 0, 0)).rolling(window=period).mean()
        loss = (-delta.where(delta < 0, 0)).rolling(window=period).mean()
        rs = gain / loss
        return 100 - (100 / (1 + rs))
    
    def _calculate_macd(self, prices, fast=12, slow=26, signal=9):
        ema_fast = prices.ewm(span=fast).mean()
        ema_slow = prices.ewm(span=slow).mean()
        macd = ema_fast - ema_slow
        macd_signal = macd.ewm(span=signal).mean()
        macd_hist = macd - macd_signal
        return macd, macd_signal, macd_hist
    
    def _calculate_bollinger(self, prices, period=20, std=2):
        sma = prices.rolling(period).mean()
        std_dev = prices.rolling(period).std()
        upper = sma + (std * std_dev)
        lower = sma - (std * std_dev)
        return upper, sma, lower
```

### 2.3 问题：模型架构可改进

**当前状态：** 使用基础的两层LSTM，没有注意力机制。

**建议优化：** 添加Attention层和更复杂的架构

```python
# strategy/models/attention_lstm.py
import tensorflow as tf
from tensorflow.keras.layers import (
    LSTM, Dense, Dropout, Input, 
    MultiHeadAttention, LayerNormalization, Add
)
from tensorflow.keras.models import Model

class AttentionLSTM:
    """带注意力机制的LSTM模型"""
    
    def __init__(self, config):
        self.config = config
        self.lstm_units = config.get("strategy.training.model_architecture.lstm_units", [128, 64])
        self.dropout_rate = config.get("strategy.training.model_architecture.dropout_rate", 0.3)
        self.learning_rate = config.get("strategy.training.model_architecture.learning_rate", 0.0005)
    
    def build_model(self, input_shape):
        """构建模型"""
        inputs = Input(shape=input_shape)
        
        # 第一层LSTM
        x = LSTM(self.lstm_units[0], return_sequences=True)(inputs)
        x = Dropout(self.dropout_rate)(x)
        
        # 注意力层
        attention_output = MultiHeadAttention(
            num_heads=4, 
            key_dim=self.lstm_units[0] // 4
        )(x, x)
        x = Add()([x, attention_output])
        x = LayerNormalization()(x)
        
        # 第二层LSTM
        x = LSTM(self.lstm_units[1], return_sequences=False)(x)
        x = Dropout(self.dropout_rate)(x)
        
        # 全连接层
        x = Dense(32, activation='relu')(x)
        x = Dropout(self.dropout_rate / 2)(x)
        
        # 输出层
        outputs = Dense(1, activation='linear')(x)
        
        model = Model(inputs=inputs, outputs=outputs)
        
        # 使用自定义学习率调度
        lr_schedule = tf.keras.optimizers.schedules.ExponentialDecay(
            initial_learning_rate=self.learning_rate,
            decay_steps=1000,
            decay_rate=0.96
        )
        
        model.compile(
            optimizer=tf.keras.optimizers.Adam(learning_rate=lr_schedule),
            loss='huber',  # 使用Huber Loss代替MSE，对异常值更鲁棒
            metrics=['mae']
        )
        
        return model
```

### 3.2 问题：信号生成过于频繁

**当前状态：** 从日志可以看到每分钟甚至更频繁地生成信号，导致大量订单。

**建议优化：** 添加信号去重和冷却机制

```python
# strategy/signal_filter.py
from datetime import datetime, timedelta
from collections import defaultdict

class SignalFilter:
    """信号过滤器 - 防止重复信号和过度交易"""
    
    def __init__(self, config):
        self.config = config
        self.signal_history = defaultdict(list)  # {symbol: [(timestamp, signal_type), ...]}
        self.cooldown_seconds = config.get("strategy.signal_cooldown", 600)  # 10分钟冷却
        self.max_signals_per_day = config.get("strategy.max_signals_per_day", 10)
        self.price_change_threshold = config.get("strategy.price_change_threshold", 0.01)  # 1%价格变化才生成新信号
        
        self.last_signal_price = {}  # {symbol: price}
    
    def should_emit_signal(self, signal) -> tuple[bool, str]:
        """判断是否应该发出信号"""
        symbol = signal.symbol
        now = datetime.now()
        
        # 1. 检查冷却时间
        if symbol in self.signal_history:
            recent_signals = [
                (ts, st) for ts, st in self.signal_history[symbol]
                if now - ts < timedelta(seconds=self.cooldown_seconds)
            ]
            
            if recent_signals:
                last_ts, last_type = recent_signals[-1]
                # 相同类型信号在冷却期内
                if last_type == signal.signal_type.value:
                    return False, f"冷却期内: 距上次{last_type}信号仅{(now-last_ts).seconds}秒"
        
        # 2. 检查每日信号数量限制
        today_start = now.replace(hour=0, minute=0, second=0, microsecond=0)
        today_signals = [
            (ts, st) for ts, st in self.signal_history.get(symbol, [])
            if ts >= today_start
        ]
        if len(today_signals) >= self.max_signals_per_day:
            return False, f"已达到每日信号上限: {self.max_signals_per_day}"
        
        # 3. 检查价格变化阈值
        if symbol in self.last_signal_price:
            last_price = self.last_signal_price[symbol]
            price_change = abs(signal.price - last_price) / last_price
            if price_change < self.price_change_threshold:
                return False, f"价格变化不足: {price_change:.2%} < {self.price_change_threshold:.2%}"
        
        # 4. 检查信号置信度
        min_confidence = self.config.get("strategy.signal_processing.confidence_threshold", 0.15)
        if signal.confidence < min_confidence:
            return False, f"置信度不足: {signal.confidence:.3f} < {min_confidence}"
        
        return True, "通过所有过滤条件"
    
    def record_signal(self, signal):
        """记录已发出的信号"""
        symbol = signal.symbol
        self.signal_history[symbol].append((datetime.now(), signal.signal_type.value))
        self.last_signal_price[symbol] = signal.price
        
        # 清理过期历史
        cutoff = datetime.now() - timedelta(days=1)
        self.signal_history[symbol] = [
            (ts, st) for ts, st in self.signal_history[symbol]
            if ts > cutoff
        ]
```

```python
# strategy/strategy_ensemble.py - 添加方法
async def update_weights_from_trades(self, trade_results: List[Dict]):
    """基于实际交易结果更新策略权重"""
    
    strategy_returns = defaultdict(list)
    
    for trade in trade_results:
        strategy_name = trade.get('strategy_name', 'unknown')
        pnl = trade.get('realized_pnl', 0)
        investment = trade.get('investment', 1)
        
        if investment > 0:
            return_rate = pnl / investment
            strategy_returns[strategy_name].append(return_rate)
    
    # 计算每个策略的表现指标
    new_weights = {}
    total_score = 0
    
    for strategy_name in self.strategies.keys():
        returns = strategy_returns.get(strategy_name, [0])
        
        if len(returns) > 5:  # 至少需要5笔交易
            avg_return = np.mean(returns)
            win_rate = len([r for r in returns if r > 0]) / len(returns)
            sharpe = np.mean(returns) / (np.std(returns) + 0.0001)
            
            # 综合评分
            score = (
                avg_return * 0.3 +
                win_rate * 0.4 +
                min(sharpe, 2) / 2 * 0.3
            )
            score = max(0.1, score)  # 最低权重
        else:
            score = 0.5  # 默认权重
        
        new_weights[strategy_name] = score
        total_score += score
    
    # 归一化
    for strategy_name in new_weights:
        self.strategy_weights[strategy_name] = new_weights[strategy_name] / total_score
    
    self.logger.info(f"基于交易结果更新权重: {self.strategy_weights}")
```

```python
import asyncio
from datetime import datetime, timedelta

class ResilientTask:
    """弹性任务包装器"""
    
    def __init__(self, name: str, coro_func, logger, max_retries: int = 10):
        self.name = name
        self.coro_func = coro_func
        self.logger = logger
        self.max_retries = max_retries
        
        self.retry_count = 0
        self.last_error_time = None
        self.task = None
    
    async def run(self, *args, **kwargs):
        """运行任务，带自动重试"""
        while self.retry_count < self.max_retries:
            try:
                await self.coro_func(*args, **kwargs)
                # 正常退出，重置计数器
                self.retry_count = 0
                break
                
            except asyncio.CancelledError:
                self.logger.info(f"任务 {self.name} 被取消")
                raise
                
            except Exception as e:
                self.retry_count += 1
                self.last_error_time = datetime.now()
                
                self.logger.error(f"任务 {self.name} 出错 (重试 {self.retry_count}/{self.max_retries}): {e}")
                
                if self.retry_count >= self.max_retries:
                    self.logger.critical(f"任务 {self.name} 达到最大重试次数，停止重试")
                    raise
                
                # 指数退避
                wait_time = min(300, 2 ** self.retry_count)  # 最多等待5分钟
                self.logger.info(f"任务 {self.name} 将在 {wait_time} 秒后重试")
                await asyncio.sleep(wait_time)
    
    def start(self, *args, **kwargs):
        """启动任务"""
        self.task = asyncio.create_task(self.run(*args, **kwargs))
        return self.task
    
    def stop(self):
        """停止任务"""
        if self.task and not self.task.done():
            self.task.cancel()
```
